def preprocess_data(data_frame):
    X = data_frame.replace('unknown', np.nan)
    num_columns = X.select_dtypes(exclude='object').columns
    
    
    cat_job = CategoricalDtype(categories=[
        'blue-collar', 
        'entrepreneur', 
        'services', 
        'unemployed', 
        'technician', 
        'admin.', 
        'retired', 
        'housemaid', 
        'student', 
        'self-employed', 
        'management'
    ])

    cat_marital = CategoricalDtype(categories=[
        'single',
        'married',
        'divorced'
    ])

    cat_education = CategoricalDtype(
        categories=[
            'illiterate',
            'basic.4y',
            'basic.6y',
            'basic.9y',
            'professional.course',
            'high.school',
            'university.degree'
        ],
        ordered=True
    )

    cat_default = CategoricalDtype(categories=['yes', 'no'])
    cat_housing = CategoricalDtype(categories=['yes', 'no'])
    cat_loan = CategoricalDtype(categories=['yes', 'no'])
    cat_contact = CategoricalDtype(categories=['telephone', 'cellular'])
    cat_poutcome = CategoricalDtype(categories=['failure', 'nonexistent', 'success'])

    X.job = X.job.astype(cat_job)
    X.marital = X.marital.astype(cat_marital)
    X.education = X.education.astype(cat_education)

    X.default = X.default.astype(cat_default)
    X.housing = X.housing.astype(cat_housing)
    X.loan = X.loan.astype(cat_loan)
    X.contact = X.contact.astype(cat_contact)
    X.poutcome = X.poutcome.astype(cat_poutcome)

    cat_columns = X.select_dtypes(['category']).columns
    X[cat_columns] = X[cat_columns].apply(lambda x: x.cat.codes)
    
    month_numbers = list(range(1, 13))
    months_abbr = [ calendar.month_abbr[idx] for idx in month_numbers ]

    months_dict = dict(zip(months_abbr, month_numbers))
    X.month = X.month.apply(str.capitalize)
    X.month = X.month.map(months_dict)
    
    day_numbers = list(range(7))
    days_abbr = [ calendar.day_abbr[idx] for idx in day_numbers ]
    days_dict = dict(zip(days_abbr, day_numbers))

    X.day_of_week = X.day_of_week.apply(str.capitalize)
    X.day_of_week = X.day_of_week.map(days_dict)
        
    return X

def imput_missing_data(data):
    X_missing = data.as_matrix()
    
    imp = Imputer(missing_values=-1, strategy='most_frequent')
    
    X_imputed = imp.fit_transform(X_missing)
    
    return pd.DataFrame(data=X_imputed, columns=data.columns, index=data.index)
